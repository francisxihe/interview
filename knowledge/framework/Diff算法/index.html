React、Vue2、Vue3 的三种 Diff 算法 - 掘金

前置假设 ​

  1. tree diff: Web UI 中 DOM 节点跨层级的移动操作特别少，可以忽略不计。
  2. component diff: 不同类型的两个元素将会产生不同的树。
  3. element diff: 可以使用一个 key prop 来指示在不同的渲染中那些元素可以保持稳定。

事实上，这些假设在几乎所有的用例中都是有效的。

React 递增法 ​

React 的思路是递增法。

  1. 交叉遍历，先新列表：找出新节点在旧列表中的位置。
    1. 如果没找到，说明是新增节点。
    2. 如果找到了。
      1. 如果位置是递增的，则不移动，记录当前位置。
      2. 否则移动旧节点到当前的新位置。
  2. 交叉遍历，先旧列表：把新列表中不存在的旧节点（从 parent）删除
查看代码
js
function reactDiff(oldChildren, newChildren, parent) {
  let lastIndex = 0;
  for (let i = 0; i < newChildren.length; i++) {
    let newChild = newChildren[i];
    let find = false; // 新节点是否在旧列表中存在
    for (let j = 0; j < oldChildren.length; j++) {
      // i 是节点在新列表中的位置，j 是节点在旧列表中的位置
      let oldChild = oldChildren[j];
      if (newChild.key === oldChild.key) {
        find = true;
        patch(oldChild, newChild, parent);
        if (j < lastIndex) {
          // 移动到（新列表）前一个节点的后面
          let refNode = newChildren[i - 1].el.newSibling;
          parent.insertBefore(newChild.el, refNode);
        } else {
          // 不需要移动节点，记录当前位置，与之后的节点进行对比
          lastIndex = j;
        }
        break;
      }
    }
    if (!find) {
      // 插入新节点
      let refNode =
        i <= 0 ? oldChildren[0].el : newChildren[i - 1].el.newSibling;
      mount(newChild, parent, refNode);
    }
  }
  // 遍历旧列表，把新列表中不存在的旧节点（从 parent）删除
  for (let i = 0; i < oldChildren.length; i++) {
    let oldChild = oldChildren[i],
      key = oldChild.key,
      has = newChildren.find((item) => item.key === key);
    if (!has) parent.removeChild(oldChild.el);
  }
}

复杂度 O(n*m) ​

两次交叉遍历，时间复杂度 O(n*m)（新旧列表的数量）。

不足 ​

  1. 遍历时，频繁找索引。时间换空间，把 key 与 index 的关系维护成一个 Map，vue3diff 有做这个优化。
  2. 当有一段未改变顺序的子序列时，子序列的每个元素都移动了一遍。vue2diff 解决了这个问题。

Vue2 双端比较 ​

所谓双端比较就是新列表和旧列表两个列表的头与尾互相对比，在对比的过程中指针会逐渐向内靠拢，直到某一个列表的节点全部遍历过，对比停止。

  1. 当首尾四次对比找到可复用的节点时，直接 patch 或移动到当前的新位置。指针向内移动。
  2. 找出新节点在旧列表中的位置。
    1. 找到了，移动到当前的新位置。DOM 移动后，将旧列表中的节点改为 undefined，代表不需要进行再次对比了。当旧列表遍历到 undefind 时就跳过。
    2. 没找到，说明是新增节点
  3. 当旧列表先遍历完时，新列表中如果有剩下节点，全部按新增节点处理。
  4. 当新列表先遍历完时，旧列表中如果有剩下节点，全部删除。
查看代码
js
import mount from "../mount";
import patch from "../patch";

export default function vue2diff(oldChildren, newChildren, parent) {
  let oldStartIndex = 0,
    newStartIndex = 0,
    oldEndIndex = oldChildren.length - 1,
    newEndIndexIndex = newChildren.length - 1,
    oldStartNode = oldChildren[oldStartIndex],
    oldEndNode = oldChildren[oldEndIndex],
    newStartNode = newChildren[newStartIndex],
    newEndIndexNode = newChildren[newEndIndexIndex];
  while (oldStartIndex <= oldEndIndex && newStartIndex <= newEndIndexIndex) {
    if (oldStartNode === undefined) {
      oldStartNode = oldChildren[++oldStartIndex];
    } else if (oldEndNode === undefined) {
      oldEndNode = oldChildren[--oldEndIndex];
    } else if (oldStartNode.key === newStartNode.key) {
      patch(oldStartNode, newStartNode, parent);

      oldStartIndex++;
      newStartIndex++;
      oldStartNode = oldChildren[oldStartIndex];
      newStartNode = newChildren[newStartIndex];
    } else if (oldEndNode.key === newEndIndexNode.key) {
      patch(oldEndNode, newEndIndexNode, parent);

      oldEndIndex--;
      newEndIndexIndex--;
      oldEndNode = oldChildren[oldEndIndex];
      newEndIndexNode = newChildren[newEndIndexIndex];
    } else if (oldStartNode.key === newEndIndexNode.key) {
      patch(oldStartNode, newEndIndexNode, parent);
      parent.insertBefore(oldStartNode.el, oldEndNode.el.newSibling);
      oldStartIndex++;
      newEndIndexIndex--;
      oldStartNode = oldChildren[oldStartIndex];
      newEndIndexNode = newChildren[newEndIndexIndex];
    } else if (oldEndNode.key === newStartNode.key) {
      patch(oldEndNode, newStartNode, parent);
      parent.insertBefore(oldEndNode.el, oldStartNode.el);
      oldEndIndex--;
      newStartIndex++;
      oldEndNode = oldChildren[oldEndIndex];
      newStartNode = newChildren[newStartIndex];
    } else {
      // 找出新节点在旧列表中的位置。
      let newKey = newStartNode.key,
        oldIndex = oldChildren.findIndex(
          (child) => child && child.key === newKey
        );
      if (oldIndex === -1) {
        // 没找到，说明是新增节点
        mount(newStartNode, parent, oldStartNode.el);
      } else {
        // 找到了，移动到当前的新位置。
        let oldNode = oldChildren[oldIndex];
        patch(oldNode, newStartNode, parent);
        parent.insertBefore(oldNode.el, oldStartNode.el);
        // DOM 移动后，将旧列表中的节点改为 undefined，代表不需要进行再次对比了。
        oldChildren[oldIndex] = undefined;
      }
      newStartIndex++;
      newStartNode = newChildren[newStartIndex];
    }
  }
  if (newStartIndex > newEndIndexIndex) {
    // 当新列表先遍历完时，旧列表中如果有剩下节点，全部删除。
    while (oldStartIndex <= oldEndIndex) {
      if (!oldChildren[oldStartIndex]) {
        oldStartIndex++;
        continue;
      }
      parent.removeChild(oldChildren[oldStartIndex++].el);
    }
  } else if (oldStartIndex > oldEndIndex) {
    // 当旧列表先遍历完时，新列表中如果有剩下节点，全部按新增节点处理。
    while (newStartIndex <= newEndIndexIndex) {
      mount(newChildren[newStartIndex++], parent, prevStartNode.el);
    }
  }
}

复杂度 O(n*m) ​

只有一次交叉遍历，时间复杂度 O(n*m)

Vue3 最长递增子序列 ​

  1. 前置与后置的预处理。
    1. key 相同，直接 patch。往中间遍历，直到 key 不同，记录位置。
    2. 正反都处理一遍，得到 key 不同的部分
  2. 旧列表已经遍历完了，新列表中剩下的节点全部按新增节点处理
  3. 新列表已经遍历完了，旧列表中剩下的节点全部删除
  4. 判断是否需要移动
    1. 先根据新列表剩余的节点数量，创建一个 source 数组储存新节点在旧列表的位置，初始值填满-1。
      1. 遍历旧列表（key 不同的部分）
        1. 如果新列表中没有旧节点 或者 已经更新了全部的新节点，直接删除旧节点
        2. 找到了，记录到给 source 里：source[newIndex - newStart] = i;
          1. 如果位置是递增的，则不需要移动。
          2. 否则需要移动。
  5. 需要移动
    1. 计算出 source 的最长递增子序列，储存子序列索引到 seq = lis(source)
    2. 从后向前进行遍历 source 每一项
      1. 当前的值为-1，这说明该节点是全新的节点，接创建好 DOM 节点插入到队尾
      2. 当前的索引为最长递增子序列中的值，也就是 i （在新列表中的位置）=== seq[j]（在旧列表中的位置），说明该节点不需要移动
      3. 当前的索引不是最长递增子序列中的值，那么说明该 DOM 节点需要移动，直接将 DOM 节点插入到队尾就可以了，因为队尾是排好序的。
  6. 不需要移动，只需要判断是否有全新的节点给他添加进去就可以了

js
import mount from "../mount";
import patch from "../patch";

export default function vue3diff(oldChildren, newChildren, parent) {
  let j = 0,/* j 代表 key 不同的起始位置 */
    oldEndIndex = oldChildren.length - 1,
    newEndIndex = newChildren.length - 1,
    oldNode = oldChildren[j],
    newNode = newChildren[j];
  outer: {
    // 往中间遍历，直到 key 不同
    while (oldNode.key === newNode.key) {
      patch(oldNode, newNode, parent);
      j++;
      if (j > oldEndIndex || j > newEndIndex) break outer;
      oldNode = oldChildren[j];
      newNode = newChildren[j];
    }

    oldNode = oldChildren[oldEndIndex];
    newNode = newChildren[newEndIndex];
    // 反着也来一遍
    while (oldNode.key === newNode.key) {
      patch(oldNode, newNode, parent);
      oldEndIndex--;
      newEndIndex--;
      if (j > oldEndIndex || j > newEndIndex) break outer;
      oldNode = oldChildren[oldEndIndex];
      newNode = newChildren[newEndIndex];
    }
  }

  if (j > oldEndIndex && j <= newEndIndex) {
    // 旧列表已经遍历完了，新列表中剩下的节点全部按新增节点处理
    let newPos = newEndIndex + 1,
      refNode =
        newPos >= newChildren.length ? null : newChildren[newPos].el;
    while (j <= newEndIndex) {
      mount(newChildren[j++], parent, refNode);
    }
    return;
  } else if (j > newEndIndex) {
    // 新列表已经遍历完了，旧列表中剩下的节点全部删除
    while (j <= oldEndIndex) {
      parent.removeChild(oldChildren[j++].el);
    }
    return;
  }

  // 先根据新列表剩余的节点数量，创建一个source数组储存新节点在旧列表的位置，初始值填满-1。
  // 计算出 source 的最长递增子序列
  let newStart = j,
    oldStart = j,
    newLeft = newEndIndex - j + 1, // 新列表中剩余的节点长度
    newIndexMap = {}, // 新列表节点与index的映射
    source = new Array(newLeft).fill(-1), // 储存新节点在旧列表的位置
    patched = 0, // 已更新过的节点的数量
    lastIndex = 0,
    move = false;

  for (let i = newStart; i <= newEndIndex; i++) {
    let key = newChildren[i].key;
    newIndexMap[key] = i;
  }
  // 去旧列表找位置
  for (let i = oldStart; i <= oldEndIndex; i++) {
    let oldChild = oldChildren[i],
      oldKey = oldChild.key,
      newIndex = newIndexMap[oldKey];
    // 新列表中没有该节点 或者 已经更新了全部的新节点，直接删除旧节点
    if (patched >= newLeft || newIndex === undefined) {
      parent.removeChild(oldChild.el);
      continue;
    }
    patched++;
    // 找到对应的节点
    let newChild = newChildren[newIndex];
    patch(oldChild, newChild, parent);
    source[newIndex - newStart] = i;

    // 同 React，全部递增，无需移动
    if (newIndex < lastIndex) {
      move = true;
    } else {
      lastIndex = newIndex;
    }
  }

  if (move) {
    const seq = lis(source);
    let j = seq.length - 1; // 下一个需要处理的 seq 索引，seq[j] 代表节点在旧列表中的位置
    for (let i = newLeft - 1; i >= 0; i--) { // i 是 source 的索引，代表在新列表中的位置
      let pos = newStart + i,
        newPos = pos + 1,
        newChild = newChildren[pos],
        refNode = newChildren[newPos]?.el;
      if (source[i] === -1) {
        mount(newChild, parent, refNode);
      } else if (i !== seq[j]) {
        parent.insertBefore(newChild.el, refNode);
      } else {
        j--;
      }
    }
  } else {
    for (let i = newLeft - 1; i >= 0; i--) {
      if (source[i] === -1) {
        let pos = newStart + i,
          newPos = pos + 1,
          newChild = newChildren[pos],
          refNode = newChildren[newPos]?.el;

        mount(newChild, parent, refNode);
      }
    }
  }
}

function lis(arr) {
  let len = arr.length,
    result = [],
    dp = new Array(len).fill(1);
  for (let i = 0; i < len; i++) {
    result.push([i]);
  }

  for (let i = len - 1; i >= 0; i--) {
    let cur = arr[i],
      newIndex = undefined;
    if (cur === -1) continue;
    for (let j = i + 1; j < len; j++) {
      let new = arr[j];
      if (cur < new) {
        let max = dp[j] + 1;
        if (max > dp[i]) {
          newIndex = j;
          dp[i] = max;
        }
      }
    }
    if (newIndex !== undefined)
      result[i] = [...result[i], ...result[newIndex]];
  }
  let index = dp.reduce(
    (prev, cur, i, arr) => (cur > arr[prev] ? i : prev),
    dp.length - 1
  );
  return result[index];
}

复杂度 O(n* logn) ​

前置与后置的预处理 + 剩余节点遍历 / 计算 source，刚好是一轮完整的 O(n+m)

计算 source 的最长递增子序列 O(n* logn)

遍历 source，O(n)

整个过程由于定义了好几个 Map，没有交叉遍历

所以取最高阶 O(n* logn)

为什么要得到最长递增子序列 ​

因为我们需要一个序列作为基础的参照序列，其他未在稳定序列的节点，进行移动。

key ​

可以更高效的帮助我们判断节点的可复用性。

使用 index 作为 key，diff 算法会从头到尾依次 patch，即所有节点均未有效的复用，等于没有 diff，全部覆盖了
